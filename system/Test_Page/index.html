from turtle import *

speed(20)
pensize(5)
penup()

f=[0,1,1]
n=11
for i in range(n):
    if i<3:
        continue
    f.append(f[i-1]+f[i-2])

for i in range(n):
    if i==0:
        continue
    r = 12*f[i]
    color("black")
    for j in range(3):
        forward(r)
        right(90)
    if i>1:
        penup()
    forward(r)
    right(90)
    pendown()
    color("red")
    circle(-r,90)
